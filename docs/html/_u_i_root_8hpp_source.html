64 // Breadth-First Traversal Algorithm for UI Finding the Component //
65 std::deque<SFUI::Component*> childrenQueue;
66 childrenQueue.push_back(rootComponent.get());

68 while (!childrenQueue.empty()) {
69 SFUI::Component* currentChild = childrenQueue.front();

72 if (currentChild->componentID == componentID) {

77 for (const auto& currentChildChild : currentChildChildren) {
78 childrenQueue.push_back(currentChildChild.get());
